Cleaning float data¶

Which columns contained ratio data? We need to convert these to numeric types. Let's define another function since we have two ratio variables. Here we replace values that can't be converted with NaN (not a number).

In [16]:
DEFAULT_VALUE = np.nan

def clean(df, column_key, convert_function, default_value):
    for i in range(len(df)):
        val = df.loc[i, column_key]  ## val = df.iloc[i, df.columns.get_loc(column_key)]
        try:
            df.loc[i, column_key] = convert_function(val)
        except (ValueError, TypeError):
            print(f'Replacing {val} with {default_value} in column {column_key} at row {i}')
            df.loc[i, column_key] = default_value

# Usage:
clean(df, BACKGROUND_YEARS_PROFESSIONAL, float, DEFAULT_VALUE)
clean(df, BACKGROUND_YEARS_PROGRAMMING, float, DEFAULT_VALUE)

EXERCISE 4: Visualisation with matplotlib¶

Making a bar chart¶

matplotlib provides functionality for creating various plots. Let's start with a frequency polygon. Note the line %matplotlib inline in is important in Jupyter.

In [24]:
IMPORT_KEYS = ['1', '2', '3', '4', '5']

def make_importance_plot(df, column_key, title):
    counts = Counter(df[column_key].dropna().astype(str))
    d = OrderedDict((k, counts.get(k, 0)) for k in IMPORT_KEYS)
    xs = [i + 0.5 for i, _ in enumerate(IMPORT_KEYS)]
    #We offset the index by 0.5 to center the bars nicely on the x-axis ticks.
    #Without +0.5, bars would start at 0, 1, 2, etc., which can make the labels or bars misaligned visually.
    #With +0.5, bars are centered at 0.5, 1.5, 2.5, ..., which matches the x-axis tick marks labeled as '1', '2', '3', etc.
    plt.bar(xs, d.values(), width=0.5)

def thresholdCap(common_list: List[Tuple], grouping_category_name: str = 'other', threshold: int = 1) -> List[Tuple]:
    """
    Aggregates elements in a list of tuples that fall below or equal to a specified threshold into a single category.

    Args:
        common_list (List[Tuple]): A list of (label, count) tuples, typically from Counter.most_common(), where the second element is the numeric value to be thresholded.

        grouping_category_name (str): The name of the category to group low-value items into. Defaults to 'other'.

        threshold (int): The value threshold for grouping. Defaults to 1.

    Returns:
        List[Tuple]: A list containing tuples above the threshold and a single tuple representing the sum of those below.
    """
    # Determine upper and lower lists categories to reduce the number of short bars in the plot
    upper_list = list(filter(lambda x: x[1] > threshold, common_list))
    lower_list = list(filter(lambda x: x[1] <= threshold, common_list))

    # Calculate the total count of items below the threshold
    total_under_thresh = reduce(lambda x, y: x + y, (map(lambda x: x[1], lower_list)))
    
    # Create a single tuple for the lower category
    lower_category = [(grouping_category_name, total_under_thresh)]
    
    # Add the lower category to the upper list
    upper_list = upper_list + lower_category
    return upper_list

    plt.xticks(xs, [name[:50] for name, _ in values], rotation = 45, fontsize = 9, horizontalalignment = 'right')
    plt.title(title)
    plt.show()

def plot_experience_histogram(data, column_key, title, bin_size=2, range_max=22):
    # Drop missing values
    values = data[column_key].dropna()

    # Compute bins based on bin size
    bins = np.arange(0, range_max + bin_size, bin_size)

    # Get histogram data
    freqs, bin_edges = np.histogram(values, bins=bins)

    # Format bin labels
    bin_labels = [f"[{int(bin_edges[i])}–{int(bin_edges[i+1])})" for i in np.arange(len(freqs))]

    # Plot
    plt.figure(figsize=(10, 5))
    plt.bar(bin_labels, freqs, width=0.8, color='lightgreen')
    plt.xticks(rotation=45)
    plt.ylabel("Number of Responses")
    plt.title(title)
    plt.tight_layout()

import pandas as pd

def remove_outliers_iqr(df, column):
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    # Keep only rows where values are within the bounds
    return df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]

# Remove outliers from both columns step-by-step
df_clean = remove_outliers_iqr(df_clean, BACKGROUND_YEARS_PROFESSIONAL)
df_clean = remove_outliers_iqr(df_clean, BACKGROUND_YEARS_PROGRAMMING)
